import { handleEvent } from './assessment/diagnostics';
import { questionResponseIds } from './assessment/questions';

/**
 * Learnosity Toolkit constructor.
 * @module App
 */

const state = {};

/**
 * Constructor for Learnosity Toolkit.
 * @since 0.1.0
 * @param {object} app - Items API app instance
 * @example
 * // Declare your variable and set with the LearnosityItems.init() method
 * const itemsApp = LearnosityItems.init(initializationObject);
 *
 * // Pass that app instance to the constructor
 * import { LT } from '[path/to/tookit/index]';
 * LT.init(itemsApp);
 *
 * // Put in whatever scope you need, eg global
 * window.LT = LT;
 */
export function init(app) {
    state.app = app;
    setupListeners();
}

/**
 * Returns an Items API app instance that the host page declared.
 * @since 0.1.0
 * @returns {object}
 */
export function appInstance() {
    return state.app;
}

/**
 * Sets up item and question listeners to pass to diagnostics.
 */
function setupListeners() {
    // Sends all Items and Assess API events for handling.
    state.app.on('all', e => {
        handleEvent(e);
    });

    // Sends Questions API `changed` event for handling.
    state.app.on('item:load', e => {
        let response_ids = questionResponseIds();
        response_ids.forEach(id => {
            state.app.question(id).on('changed', ev => {
                handleEvent('changed');
            });
        });
    });
}
